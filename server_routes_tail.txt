  // Body: { turmaId, disciplina, trimestre, anoLetivo }
  // Converte nivel → escala 1-5: muito_bom=5, bom=4, satisfatorio=3, insuficiente=2
  // MAC_final = MAC_sumativo×(1-percFormativa/100) + notaFormativa×(percFormativa/100)
  app.post("/api/avaliacoes-formativas/calcular", requireAuth, async (req: Request, res: Response) => {
    try {
      const b = requireBodyObject(req);
      const { turmaId, disciplina, trimestre, anoLetivo } = b as Record<string, unknown>;
      if (!turmaId || !disciplina || !trimestre || !anoLetivo) {
        return json(res, 400, { error: "turmaId, disciplina, trimestre e anoLetivo são obrigatórios" });
      }
      const NIVEL_ESCALA: Record<string, number> = {
        muito_bom: 5, bom: 4, satisfatorio: 3, insuficiente: 2
      };
      // Calcular média por aluno
      const rows = await query<{ alunoId: string; media: string }>(
        `SELECT "alunoId",
                AVG(CASE nivel
                      WHEN 'muito_bom'    THEN 5
                      WHEN 'bom'          THEN 4
                      WHEN 'satisfatorio' THEN 3
                      WHEN 'insuficiente' THEN 2
                      ELSE 3
                    END) as media
         FROM public.avaliacoes_formativas
         WHERE "turmaId"=$1 AND disciplina=$2 AND trimestre=$3 AND "anoLetivo"=$4
         GROUP BY "alunoId"`,
        [turmaId, disciplina, parseInt(String(trimestre)), anoLetivo]
      );
      if (!rows.length) return json(res, 200, { atualizados: 0, resultados: [] });
      const resultados: { alunoId: string; notaFormativa: number; atualizado: boolean }[] = [];
      for (const row of rows) {
        const notaFormativa = Math.round(parseFloat(row.media) * 10) / 10;
        // Garantir registo na tabela notas; se não existir, o professor ainda não abriu a pauta — ignorar
        const updated = await query<{ id: string }>(
          `UPDATE public.notas SET "notaFormativa"=$1
           WHERE "turmaId"=$2 AND disciplina=$3 AND trimestre=$4 AND "anoLetivo"=$5 AND "alunoId"=$6
           RETURNING id`,
          [notaFormativa, turmaId, disciplina, parseInt(String(trimestre)), anoLetivo, row.alunoId]
        );
        resultados.push({ alunoId: row.alunoId, notaFormativa, atualizado: updated.length > 0 });
      }
      json(res, 200, { atualizados: resultados.filter(r => r.atualizado).length, resultados });
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

             req.jwtUser?.userId || null, req.jwtUser?.email || '']
          );
          if (ins.length) saved.push(ins[0]);
        }
      }
      json(res, 200, { ok: true, count: saved.length, registos: saved });
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

  // PUT /api/diagnostica/:id — actualizar registo individual
  app.put("/api/diagnostica/:id", requireAuth, async (req: Request, res: Response) => {
    try {
      const b = requireBodyObject(req);
      const NIVEIS = ['nao_satisfaz', 'satisfaz', 'bom', 'muito_bom'];
      const nivel = NIVEIS.includes(b.nivel as string) ? b.nivel as string : 'satisfaz';
      const rows = await query<JsonObject>(
        `UPDATE public.avaliacoes_diagnosticas
         SET nivel=$1, nota=$2, observacoes=$3, "registadoPorId"=$4, "registadoPorNome"=$5, "registadoEm"=now()
         WHERE id=$6 RETURNING *`,
        [nivel, b.nota ?? null, b.observacoes || '',
         req.jwtUser?.userId || null, req.jwtUser?.email || '', req.params.id]
      );
      if (!rows.length) return json(res, 404, { error: 'Registo não encontrado.' });
      json(res, 200, rows[0]);
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

  // DELETE /api/diagnostica/:id — remover registo
  app.delete("/api/diagnostica/:id", requireAuth, async (req: Request, res: Response) => {
    try {
      await query(`DELETE FROM public.avaliacoes_diagnosticas WHERE id=$1`, [req.params.id]);
      json(res, 200, { ok: true });
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

  // GET /api/diagnostica/resumo/:turmaId — resumo por turma
  app.get("/api/diagnostica/resumo/:turmaId", requireAuth, async (req: Request, res: Response) => {
    try {
      const { anoLetivo, trimestre } = req.query as Record<string, string>;
      const params: unknown[] = [req.params.turmaId];
      let where = `WHERE d."turmaId"=$1`;
      if (anoLetivo) { params.push(anoLetivo); where += ` AND d."anoLetivo"=${params.length}`; }
      if (trimestre) { params.push(parseInt(trimestre)); where += ` AND d.trimestre=${params.length}`; }
      const rows = await query<JsonObject>(
        `SELECT d."disciplinaNome", d.trimestre,
           COUNT(*)::int as total,
           COUNT(*) FILTER (WHERE d.nivel='muito_bom')::int as "muitoBom",
           COUNT(*) FILTER (WHERE d.nivel='bom')::int as "bom",
           COUNT(*) FILTER (WHERE d.nivel='satisfaz')::int as "satisfaz",
           COUNT(*) FILTER (WHERE d.nivel='nao_satisfaz')::int as "naoSatisfaz",
           ROUND(AVG(d.nota)::numeric,1) as "mediaNotas"
         FROM public.avaliacoes_diagnosticas d
         ${where}
         GROUP BY d."disciplinaNome", d.trimestre
         ORDER BY d.trimestre, d."disciplinaNome"`,
        params
      );
      json(res, 200, rows);
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

  // GET /api/diagnostica/relatorio — resumo global por turma/disciplina/trimestre (admin/director)
  app.get("/api/diagnostica/relatorio", requireAuth, async (req: Request, res: Response) => {
    try {
      const { anoLetivo, turmaId, trimestre } = req.query as Record<string, string>;
      const params: unknown[] = [];
      let where = "WHERE 1=1";
      if (anoLetivo)  { params.push(anoLetivo);          where += ` AND d."anoLetivo"=$${params.length}`; }
      if (turmaId)    { params.push(turmaId);            where += ` AND d."turmaId"=$${params.length}`; }
      if (trimestre)  { params.push(parseInt(trimestre)); where += ` AND d.trimestre=$${params.length}`; }
      const rows = await query<JsonObject>(
        `SELECT d."turmaId", t.nome as "turmaNome", d."disciplinaNome", d.trimestre,
           COUNT(*)::int as total,
           COUNT(*) FILTER (WHERE d.nivel='muito_bom')::int  as "muitoBom",
           COUNT(*) FILTER (WHERE d.nivel='bom')::int        as "bom",
           COUNT(*) FILTER (WHERE d.nivel='satisfaz')::int   as "satisfaz",
           COUNT(*) FILTER (WHERE d.nivel='nao_satisfaz')::int as "naoSatisfaz",
           ROUND(AVG(d.nota)::numeric,1) as "mediaNotas"
         FROM public.avaliacoes_diagnosticas d
         LEFT JOIN public.turmas t ON t.id = d."turmaId"
         ${where}
         GROUP BY d."turmaId", t.nome, d."disciplinaNome", d.trimestre
         ORDER BY t.nome, d.trimestre, d."disciplinaNome"`,
        params
      );
      json(res, 200, rows);
    } catch (e) { json(res, 500, { error: (e as Error).message }); }
  });

  const httpServer = createServer(app);

  return httpServer;
}
